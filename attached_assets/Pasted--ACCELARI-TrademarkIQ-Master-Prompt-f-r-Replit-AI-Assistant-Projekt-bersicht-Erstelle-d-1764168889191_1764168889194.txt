### 3. Button-Komponente (ACCELARI-Klassen)

```tsx
// components/ui/Button.tsx

interface ButtonProps {
  variant: 'primary' | 'secondary' | 'link';
  href?: string;
  children: React.ReactNode;
  className?: string;
}

export function Button({ variant, href, children, className }: ButtonProps) {
  const baseClass = variant === 'link' 
    ? 's-link' 
    : `s-button ${variant === 'secondary' ? 's-button-two' : ''}`;
  
  return (
    <a href={href} className={`${baseClass} ${className || ''}`}>
      {children}
      {variant === 'link' && (
        <svg width="11" height="11" viewBox="0 0 11 11">
          <path d="M3 1l5 4.5L3 10" stroke="currentColor" fill="none"/>
        </svg>
      )}
    </a>
  );
}

/* Verwendung */
```

```tsx
  padding?: 'sm' | 'md' | 'lg' | 'xl';
  children: React.ReactNode;
}

  white: '',
  gray: 'background: #F5F5F5;',
  teal: 'background: #0D9488; color: #fff;',

  sm: 'padding-top: 40px; padding-bottom: 40px;',
  md: 'padding-top: 60px; padding-bottom: 60px;',

};

export function Section({ background = 'white', padding = 'lg', children }: SectionProps) {
  return (
    <section className="s-section" style={`${backgrounds[background]} ${paddings[padding]}`}>

  );
}
```
